 36class Logger:
 37    """
 38    Класс для управления выводом сообщений.
 39    Обеспечивает одновременный вывод в консоль (с цветом) и в лог-файл (без цвета).
 40    """
 41
 42    def __init__(self):
 43        """Инициализирует логгер с настройками по умолчанию."""
 44        self.debug_mode = False
 45        self.color_enabled = sys.stdout.isatty()
 46        self.log_file_path = None
 47        self.log_file = None
 48
 49    def reconfigure(self, debug_mode=False, color_enabled=True, log_file=None):
 50        """
 51        Переконфигурирует логгер на основе параметров командной строки.
 52
 53        :param debug_mode: Включить ли отладочный вывод.
 54        :param color_enabled: Включить ли цветной вывод.
 55        :param log_file: Путь к файлу для логирования.
 56        """
 57        self.debug_mode = debug_mode
 58        self.color_enabled = color_enabled and sys.stdout.isatty()
 59        if self.log_file_path != log_file:
 60            if self.log_file:
 61                self.log_file.close()
 62            self.log_file_path = log_file
 63            if self.log_file_path:
 64                try:
 65                    self.log_file = open(self.log_file_path, 'a', encoding='utf-8')
 66                except IOError as e:
 67                    sys.stderr.write(f"ПРЕДУПРЕЖДЕНИЕ: Не удалось открыть лог-файл '{self.log_file_path}': {e}\n")
 68                    self.log_file = None
 69
 70    @staticmethod
 71    def _strip_ansi(text: str) -> str:
 72        """Удаляет ANSI-коды из строки."""
 73        return re.sub(r'\033\[[0-9;]*m', '', text)
 74
 75    def _write(self, message: str):
 76        """Основной метод записи в stdout и в файл."""
 77        print(message)
 78        if self.log_file:
 79            self.log_file.write(Logger._strip_ansi(message) + '\n')
 80            self.log_file.flush()
 81
 82    def _colorize(self, text: str, color: str) -> str:
 83        """Оборачивает текст в ANSI-коды цвета."""
 84        return f"{color}{text}{AnsiColors.ENDC}" if self.color_enabled else text
 85
 86    def info(self, message: str):
 87        """Выводит информационное сообщение."""
 88        self._write(message)
 89
 90    def success(self, message: str):
 91        """Выводит сообщение об успехе (зеленым)."""
 92        self._write(self._colorize(message, AnsiColors.OKGREEN))
 93
 94    def warning(self, message: str):
 95        """Выводит предупреждение (желтым)."""
 96        self._write(self._colorize(message, AnsiColors.WARNING))
 97
 98    def error(self, message: str, to_stderr=True):
 99        """Выводит сообщение об ошибке (красным)."""
100        formatted_message = self._colorize(message, AnsiColors.FAIL)
101        if to_stderr:
102            if self.log_file:
103                self.log_file.write(Logger._strip_ansi(formatted_message) + '\n')
104                self.log_file.flush()
105            print(formatted_message, file=sys.stderr)
106        else:
107            self._write(formatted_message)
108
109    def debug(self, message: str):
110        """Выводит отладочное сообщение."""
111        if self.debug_mode:
112            self._write(self._colorize(f"[DEBUG] {message}", AnsiColors.OKCYAN))
113
114    def header(self, message: str):
115        """Выводит заголовок (жирным)."""
116        self._write(self._colorize(message, AnsiColors.BOLD))
117
118    def __del__(self):
119        """Закрывает файл лога при уничтожении объекта."""
120        if self.log_file:
121            self.log_file.close()

172def check_squid_version(args):
173    """
174    Проверяет версию Squid, используя существующую функцию run_shell_command.
175    Также проверяет наличие флага --enable-icap-client при компиляции Squid.
176
177    Выполняет команду "squid --version" и ищет версию Squid и флаг --enable-icap-client.
178    Если они не найдены, выводит сообщение об ошибке и завершает скрипт.  
179    """
180
181    output = run_shell_command(['squid', '--version'], "Проверка совместимости установленной версии Squid с Dr.Web")
182
183    """ Берет первую строку вывода и ищет номер версии"""
184    version_line = output.split("\n")[0]
185    for ch in range(len(version_line)):
186        if version_line[ch].isdigit():
187            version = version_line[ch:].strip()
188            break
189    
190    """ Проверяет поддерживается ли данная версия """
191    if int(version[0]) < 3:
192        logger.error("Ошибка: Версии ниже 3.0 не поддерживаются.")
193        sys.exit(1)
194
195    """ Проверяет наличие флага --enable-icap-client """
196    if "--enable-icap-client" not in output:
197        logger.error("Ошибка: Установленная версия Squid была скомпилирона без флага --enable-icap-client.")
198        sys.exit(1)
199
200    if args.command == 'setup':
201        if args.with_ssl:
202            if "--with-openssl" not in output or "--enable-ssl-crtd" not in output:
203                logger.error("Ошибка: Установленная версия Squid была скомпилирона без поддержки разбора HTTPS. Перекомпилируйте squid с флагами --with-openssl и --enable-ssl-crtd.")
204                sys.exit(1)         
205     
206    return version

239def run_shell_command(command: list, title: str = ""):
240    """
241    Выполняет внешнюю команду в оболочке ОС.
242
243    :param command: Список, содержащий команду и ее аргументы.
244    :param title: Описание действия для логирования.
245    :raises RuntimeError: Если команда завершается с ошибкой.
246    :return: Строка с stdout выполненной команды.
247    """
248    if title:
249        logger.debug(f"--- {title} ---")
250    logger.debug(f"Выполнение shell-команды: {' '.join(command)}")
251    try:
252        process = subprocess.run(
253            command, capture_output=True, text=True, check=True, timeout=300
254        )
255        if process.stdout.strip():
256            logger.debug(f"STDOUT:\n{process.stdout.strip()}")
257        return process.stdout.strip()
258    except subprocess.CalledProcessError as e:
259        error_message = (
260            f"Команда '{' '.join(e.cmd)}' завершилась с ошибкой (код {e.returncode}).\n"
261            f"STDERR: {e.stderr.strip()}"
262        )
263        raise RuntimeError(error_message)
264    except FileNotFoundError:
265        raise RuntimeError(f"Команда '{command[0]}' не найдена. Убедитесь, что утилита установлена и доступна в PATH.")
266    except Exception as e:
267        raise RuntimeError(f"Непредвиденная ошибка при выполнении команды: {e}")
